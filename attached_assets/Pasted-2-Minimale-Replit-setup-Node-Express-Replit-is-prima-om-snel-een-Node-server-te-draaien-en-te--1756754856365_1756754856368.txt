Stap C — index.js

import express from "express";

const app = express();
app.use(express.json());

app.post("/api/mindstudio/run", async (req, res) => {
  try {
    // Client stuurt { variables: {...} } mee
    const variables = req.body?.variables ?? {};

    const resp = await fetch("https://api.mindstudio.ai/developer/v2/apps/run", {
      method: "POST",
      headers: {
        "Authorization": `Bearer ${process.env.MINDSTUDIO_API_KEY}`,
        "Content-Type": "application/json"
      },
      body: JSON.stringify({
        workerId: process.env.MINDSTUDIO_WORKER_ID,
        variables,
        workflow: process.env.MINDSTUDIO_WORKFLOW || "Main.flow",
        version: process.env.MINDSTUDIO_VERSION || "published"
      })
    });

    const data = await resp.json();
    res.status(resp.ok ? 200 : resp.status).json(data);
  } catch (err) {
    console.error(err);
    res.status(500).json({ error: "Mindstudio call failed" });
  }
});

app.get("/", (_, res) => res.send("Replit x Mindstudio OK"));
const port = process.env.PORT || 3000;
app.listen(port, () => console.log(`Server on :${port}`));


Waarom via je eigen endpoint?
Je leakt dan je Mindstudio API key niet naar de browser; server-to-server is de juiste manier. (Algemene best practice.)
